Setup

# One-time setup
curl -sL https://opencode.calavia.org/setup.sh | bash

# Add tokens to your profile
export GITHUB_TOKEN="ghp_..."

# Optional: Context7 for current docs
export CONTEXT7_API_KEY="ctx7_..."

# Optional: Expert config validator
export EXPERT_CONFIG_DIR="~/.config/opencode"

See Token Setup Guide for bot account and HUMAN token configuration.
